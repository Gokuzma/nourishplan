```sql
-- grocery_lists: one snapshot per household per week
CREATE TABLE IF NOT EXISTS public.grocery_lists (
  id           uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  household_id uuid NOT NULL REFERENCES public.households(id) ON DELETE CASCADE,
  week_start   date NOT NULL,
  generated_at timestamptz NOT NULL DEFAULT now(),
  generated_by uuid NOT NULL REFERENCES auth.users(id),
  UNIQUE (household_id, week_start)
);

ALTER TABLE public.grocery_lists ENABLE ROW LEVEL SECURITY;

CREATE POLICY "household members read grocery_lists"
  ON public.grocery_lists FOR SELECT TO authenticated
  USING (household_id = get_user_household_id());

CREATE POLICY "household members insert grocery_lists"
  ON public.grocery_lists FOR INSERT TO authenticated
  WITH CHECK (household_id = get_user_household_id());

CREATE POLICY "household members update grocery_lists"
  ON public.grocery_lists FOR UPDATE TO authenticated
  USING (household_id = get_user_household_id());

CREATE POLICY "household members delete grocery_lists"
  ON public.grocery_lists FOR DELETE TO authenticated
  USING (household_id = get_user_household_id());

-- grocery_items: line items (generated + manual)
CREATE TABLE IF NOT EXISTS public.grocery_items (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  list_id         uuid NOT NULL REFERENCES public.grocery_lists(id) ON DELETE CASCADE,
  household_id    uuid NOT NULL REFERENCES public.households(id) ON DELETE CASCADE,
  food_name       text NOT NULL,
  food_id         text,
  quantity         numeric,
  unit            text,
  category        text NOT NULL DEFAULT 'Other',
  category_source text NOT NULL DEFAULT 'auto',
  is_checked      boolean NOT NULL DEFAULT false,
  checked_by      uuid REFERENCES auth.users(id),
  checked_at      timestamptz,
  is_manual       boolean NOT NULL DEFAULT false,
  is_staple_restock boolean NOT NULL DEFAULT false,
  estimated_cost  numeric,
  notes           text,
  created_at      timestamptz NOT NULL DEFAULT now(),
  updated_at      timestamptz NOT NULL DEFAULT now()
);

ALTER TABLE public.grocery_items ENABLE ROW LEVEL SECURITY;

CREATE POLICY "household members read grocery_items"
  ON public.grocery_items FOR SELECT TO authenticated
  USING (household_id = get_user_household_id());

CREATE POLICY "household members insert grocery_items"
  ON public.grocery_items FOR INSERT TO authenticated
  WITH CHECK (household_id = get_user_household_id());

CREATE POLICY "household members update grocery_items"
  ON public.grocery_items FOR UPDATE TO authenticated
  USING (household_id = get_user_household_id());

CREATE POLICY "household members delete grocery_items"
  ON public.grocery_items FOR DELETE TO authenticated
  USING (household_id = get_user_household_id());

-- Enable Supabase realtime for grocery_items (required for live sync per D-14)
ALTER PUBLICATION supabase_realtime ADD TABLE public.grocery_items;

-- set_updated_at trigger on grocery_items (reuse existing function from foundation migration)
CREATE TRIGGER set_grocery_items_updated_at
  BEFORE UPDATE ON public.grocery_items
  FOR EACH ROW EXECUTE FUNCTION set_updated_at();
```

Create `src/utils/groceryGeneration.ts` with these pure functions (no hooks, no Supabase calls):

**STORE_CATEGORIES constant:**
```typescript
export const STORE_CATEGORIES = [
  'Produce', 'Dairy', 'Meat & Seafood', 'Bakery', 'Pantry/Dry Goods',
  'Frozen', 'Beverages', 'Condiments & Spices', 'Snacks', 'Other',
] as const
```

**CATEGORY_KEYWORDS map:**
A Record<string, string[]> mapping category names to lowercase food name keywords. Include ~8-12 keywords per category. Default to 'Other' when no match. Categories per D-09.

**`aggregateIngredients` function:**

Input: An array of "resolved slot data" objects — each slot with its meal_items, and for recipe-type meal_items, their recipe_ingredients (already fetched by the caller). Also takes `cookedRecipeIds: Set<string>` (recipe IDs from spend_logs with source='cook' in current week) for D-03 exclusion.

Processing:
1. Filter out slots where meal_id is null
2. For each slot's meal_items: if item_type='food', treat as a direct ingredient (food_id=item_id, food_name from item_name field on meal_items or fallback, quantity_grams from meal_item). If item_type='recipe', traverse recipe_ingredients recursively (depth limit 5, console.warn at limit).
3. For recipe-type items: skip the recipe entirely if its item_id is in cookedRecipeIds (D-03)
4. Aggregate by food_id (case-insensitive food_name as fallback key when food_id is null). Sum quantity_grams.

Output: `AggregatedIngredient[]` where:
```typescript
export interface AggregatedIngredient {
  food_id: string | null
  food_name: string
  total_grams: number
}
```

**`subtractInventory` function:**

Input: `AggregatedIngredient[]`, `InventoryItem[]`

Processing: For each aggregated ingredient, call `getAvailableQuantity(inventoryItems, food_id)`. If available >= total_grams, it goes to "alreadyHave". If available > 0 but < total_grams, it goes to "needToBuy" with remainder = total_grams - available, and also to "alreadyHave" for the covered portion. If available = 0, full amount goes to "needToBuy".

Output: `{ needToBuy: GroceryItemDraft[], alreadyHave: GroceryItemDraft[] }` where:
```typescript
export interface GroceryItemDraft {
  food_id: string | null
  food_name: string
  quantity_grams: number
  covered_grams?: number  // for alreadyHave: how much inventory covers
  total_needed_grams?: number  // for alreadyHave: original total before subtraction
}
```

**`assignCategories` function:**

Input: `GroceryItemDraft[]`, `previousItems: { food_id: string | null; food_name: string; category: string; category_source: string }[]`

Processing: For each item, first check previousItems for matching food_id (or food_name) with category_source='user' — if found, use that category + source='user'. Otherwise, match food_name keywords against CATEGORY_KEYWORDS map. Default to 'Other' with source='auto'.

Output: items with `category: string` and `category_source: 'auto' | 'user'` added.

**`formatDisplayQuantity` function:**

Input: `quantity_grams: number, isLiquid?: boolean`

Processing: if >= 1000, display as kg (or L for liquid). Otherwise display as g (or ml). Round to 1 decimal place.

Output: `{ display_quantity: number, display_unit: string }` e.g. `{ display_quantity: 1.5, display_unit: 'kg' }`

**`addRestockStaples` function:**

Input: `existingFoodIds: Set<string>`, `inventoryItems: InventoryItem[]`

Processing: Call `getLowStockStaples(inventoryItems, 100)`. Filter out any whose food_id is already in existingFoodIds. Return as GroceryItemDraft[] with is_staple_restock=true flag.

**`computeItemCost` function:**

Input: `quantity_grams: number`, `food_id: string | null`, `foodPrices: FoodPrice[]`

Processing: Find price matching food_id in foodPrices. If found: `(quantity_grams / 100) * cost_per_100g`. If not found: return null.

Output: `number | null`
