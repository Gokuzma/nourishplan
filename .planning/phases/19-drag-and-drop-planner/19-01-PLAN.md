4. Add `useToggleLock` mutation to `src/hooks/useMealPlan.ts`. Place it after `useClearSlot`. Follow the exact pattern of `useClearSlot`:
```typescript
export function useToggleLock() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: async ({
      slotId,
      isLocked,
    }: {
      slotId: string
      isLocked: boolean
      planId: string
    }): Promise<void> => {
      const { error } = await supabase
        .from('meal_plan_slots')
        .update({ is_locked: isLocked })
        .eq('id', slotId)
      if (error) throw error
    },
    onSuccess: (_, { planId }) => {
      queryClient.invalidateQueries({ queryKey: queryKeys.mealPlan.slots(planId) })
    },
  })
}
```
