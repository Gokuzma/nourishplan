Export `useGenerationJob` polling query:
```typescript
export function useGenerationJob(jobId: string | null) {
  const { data: membership } = useHousehold()
  const householdId = membership?.household_id
  const queryClient = useQueryClient()

  return useQuery({
    queryKey: queryKeys.planGeneration.job(jobId, householdId),
    queryFn: async () => {
      const { data, error } = await supabase
        .from('plan_generations')
        .select('*')
        .eq('id', jobId!)
        .single()
      if (error) throw error
      return data as PlanGeneration
    },
    enabled: !!jobId && !!householdId,
    refetchInterval: (query) => {
      const status = query.state.data?.status
      if (status === 'done' || status === 'timeout' || status === 'error') return false
      return 2000  // poll every 2 seconds
    },
  })
}
```

Export `useLatestGeneration` query — fetches the most recent generation for the current plan week:
```typescript
export function useLatestGeneration(planId: string | undefined) {
  const { data: membership } = useHousehold()
  const householdId = membership?.household_id

  return useQuery({
    queryKey: ['plan-generation', householdId, planId, 'latest'] as const,
    queryFn: async () => {
      const { data, error } = await supabase
        .from('plan_generations')
        .select('*')
        .eq('plan_id', planId!)
        .order('created_at', { ascending: false })
        .limit(1)
        .maybeSingle()
      if (error) throw error
      return data as PlanGeneration | null
    },
    enabled: !!planId && !!householdId,
  })
}
```

  return useMutation({

        body: {

          priorityOrder: params.priorityOrder,
          singleSlot: { dayIndex: params.dayIndex, slotName: params.slotName },

      })
      if (error) throw error
      if (!data?.success) throw new Error(data?.error || 'Suggestion failed')
      return data
    },
    onSuccess: () => {
      // Invalidate meal plan slots to refresh the grid
      queryClient.invalidateQueries({ queryKey: ['meal-plan-slots'] })
    },
  })
}

```typescript
export function useNutritionGaps(planId: string | undefined) {
  const { data: membership } = useHousehold()
  const householdId = membership?.household_id
  const { data: slots } = useMealPlanSlots(planId)
  const { data: targets } = useNutritionTargets()
  const { data: members } = useHouseholdMembers()
  const { data: profiles } = useMemberProfiles()
  const { data: latestGen } = useLatestGeneration(planId)

  const gaps = useMemo(() => {
    if (!slots || !targets || !members) return []
    // Build MemberIdentity array from members + profiles
    const memberIdentities: MemberIdentity[] = [
      ...(members || []).map(m => ({ id: m.user_id, type: 'user' as const, name: m.display_name || 'Member' })),
      ...(profiles || []).map(p => ({ id: p.id, type: 'profile' as const, name: p.name })),
    ]
    return calcWeeklyGaps(slots, targets, memberIdentities)
  }, [slots, targets, members, profiles])

  return { gaps, hasGeneration: !!latestGen, latestGeneration: latestGen }
}
```

Note: The `members` from `useHouseholdMembers` may not have `display_name` directly — read the hook to see the actual return shape and adjust the mapping accordingly. May need to join with profiles.
