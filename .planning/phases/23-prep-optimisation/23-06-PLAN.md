```typescript
import { useState, useCallback, type ReactNode } from 'react'
import { useNavigate } from 'react-router-dom'
import { CookProgressBar } from './CookProgressBar'
import { MultiMealSwitcher } from './MultiMealSwitcher'
import type { CookSession } from '../../types/database'

interface CookModeShellProps {
  mealName: string
  subtitle: string
  completedSteps: number
  totalSteps: number
  activeSessions: CookSession[]
  currentSessionId: string | undefined
  onSessionSwitch: (sessionId: string) => void
  footer: ReactNode
  children: ReactNode
}

export function CookModeShell({
  mealName, subtitle, completedSteps, totalSteps,
  activeSessions, currentSessionId, onSessionSwitch,
  footer, children,
}: CookModeShellProps) {
  const navigate = useNavigate()
  const [showExitConfirm, setShowExitConfirm] = useState(false)
  const pct = totalSteps > 0 ? Math.round((completedSteps / totalSteps) * 100) : 0

  const handleBack = useCallback(() => {
    if (completedSteps > 0 && completedSteps < totalSteps) {
      setShowExitConfirm(true)
    } else {
      navigate(-1)
    }
  }, [completedSteps, totalSteps, navigate])

  return (
    <div className="min-h-screen flex flex-col bg-background font-sans">
      {/* Top bar */}
      <header className="sticky top-0 z-40 bg-surface border-b border-accent/20 px-4 py-3 flex items-center justify-between">
        <button
          onClick={handleBack}
          className="p-1 text-text/60 hover:text-text"
          aria-label="Exit cook mode"
        >
          <svg width="16" height="16" viewBox="0 0 16 16" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round"><path d="M10 3L5 8l5 5"/></svg>
        </button>
        <div className="flex-1 text-center min-w-0 px-2">
          <div className="text-sm font-semibold text-text truncate">{mealName}</div>
          <div className="text-xs text-text/50">{subtitle}</div>
        </div>
        {activeSessions.length > 1 ? (
          <MultiMealSwitcher
            sessions={activeSessions}
            currentSessionId={currentSessionId}
            onSwitch={onSessionSwitch}
          />
        ) : (
          <div className="w-6" />
        )}
      </header>

      {/* Progress bar */}
      <CookProgressBar completed={completedSteps} total={totalSteps} />

      {/* Exit confirmation overlay */}
      {showExitConfirm && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/40 backdrop-blur-sm">
          <div className="bg-surface rounded-2xl p-6 mx-4 max-w-sm shadow-xl flex flex-col gap-4">
            <p className="text-sm text-text font-sans">Exit cook mode? Progress will be saved.</p>
            <div className="flex gap-2">
              <button
                onClick={() => navigate(-1)}
                className="flex-1 bg-primary text-white rounded-[--radius-btn] px-4 py-2 text-sm font-semibold"
              >
                Exit
              </button>
              <button
                onClick={() => setShowExitConfirm(false)}
                className="flex-1 text-text/60 hover:text-text text-sm"
              >
                Keep cooking
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Body */}
      <main className="flex-1 overflow-y-auto px-4 py-4 space-y-3 pb-32">
        {children}
      </main>

      {/* Fixed footer */}
      <div className="fixed bottom-0 inset-x-0 z-40 bg-surface border-t border-accent/20 px-4 py-3">
        {footer}
      </div>
    </div>
  )
}
```

```typescript
interface CookStepPrimaryActionProps {
  hasActiveStep: boolean
  isLastStep: boolean
  activeStepHasTimer: boolean
  timerRunning: boolean
  timerDone: boolean
  allComplete: boolean
  onAction: () => void
}

export function CookStepPrimaryAction({
  hasActiveStep, isLastStep, activeStepHasTimer,
  timerRunning, timerDone, allComplete, onAction,
}: CookStepPrimaryActionProps) {
  let label: string
  let pulse = false

  if (allComplete) {
    label = 'Exit cook mode'
  } else if (isLastStep && hasActiveStep) {
    label = 'Finish cook session'
  } else if (!hasActiveStep) {
    label = 'Mark complete'
  } else if (activeStepHasTimer && !timerRunning && !timerDone) {
    label = 'Start timer'
  } else if (timerRunning) {
    label = 'Mark complete now'
  } else if (timerDone) {
    label = 'Mark complete'
    pulse = true
  } else {
    label = 'Mark complete'
  }

  // Check icon SVG for complete-style actions, play icon for timer start
  const isTimerStart = activeStepHasTimer && !timerRunning && !timerDone && !allComplete && hasActiveStep
  const icon = isTimerStart
    ? <svg width="16" height="16" viewBox="0 0 16 16" fill="currentColor"><path d="M5 3l8 5-8 5V3z"/></svg>
    : <svg width="16" height="16" viewBox="0 0 16 16" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round"><path d="M3 8l4 4 6-7"/></svg>

  return (
    <button
      onClick={onAction}
      className={`w-full bg-primary text-white rounded-[--radius-btn] h-12 text-sm font-semibold flex items-center justify-center gap-2 ${pulse ? 'animate-pulse' : ''}`}
    >
      {icon}
      {label}
    </button>
  )
}
```

```typescript
import { useState } from 'react'

interface ReheatStep {
  id: string
  text: string
}

interface ReheatSequenceCardProps {
  storageHint: 'fridge' | 'freezer'
  steps: ReheatStep[]
  onComplete: () => void
}

export function ReheatSequenceCard({ storageHint, steps, onComplete }: ReheatSequenceCardProps) {
  const [checked, setChecked] = useState<Set<string>>(new Set())

  const toggle = (id: string) => {
    setChecked(prev => {
      const next = new Set(prev)
      if (next.has(id)) next.delete(id)
      else next.add(id)
      return next
    })
  }

  return (
    <div className="bg-surface rounded-[--radius-card] p-4 flex flex-col gap-3">
      <h3 className="text-sm font-semibold text-text">
        Reheat from {storageHint}
      </h3>
      <ol className="space-y-2">
        {steps.map((step, i) => (
          <li key={step.id} className="flex items-start gap-3">
            <button
              onClick={() => toggle(step.id)}
              className={`mt-0.5 w-5 h-5 rounded border flex-shrink-0 flex items-center justify-center transition-colors ${
                checked.has(step.id)
                  ? 'bg-primary border-primary text-white'
                  : 'border-text/20 text-transparent'
              }`}
              aria-label={`Step ${i + 1}: ${checked.has(step.id) ? 'completed' : 'not completed'}`}
            >
              {checked.has(step.id) && (
                <svg width="12" height="12" viewBox="0 0 16 16" fill="none" stroke="currentColor" strokeWidth="2.5" strokeLinecap="round"><path d="M3 8l4 4 6-7"/></svg>
              )}
            </button>
            <span className={`text-sm font-sans ${checked.has(step.id) ? 'text-text/40 line-through' : 'text-text'}`}>
              {step.text}
            </span>
          </li>
        ))}
      </ol>
      <button
        onClick={onComplete}
        className="w-full bg-primary text-white rounded-[--radius-btn] h-12 text-sm font-semibold flex items-center justify-center gap-2 mt-2"
      >
        Done reheating
      </button>
    </div>
  )
}
```

```typescript
import { useState } from 'react'

interface CookStepTimerProps {
  secondsRemaining: number
  running: boolean
  done: boolean
  onStart: () => void
  onPause: () => void
  onReset: () => void
  onSkip: () => void
  onComplete: () => void
}

export function CookStepTimer({
  secondsRemaining, running, done,
  onStart, onPause, onReset, onSkip, onComplete,
}: CookStepTimerProps) {
  const [showSkipConfirm, setShowSkipConfirm] = useState(false)
  const isWarning = secondsRemaining > 0 && secondsRemaining <= 60
  const minutes = Math.floor(Math.max(0, secondsRemaining) / 60)
  const seconds = Math.max(0, secondsRemaining) % 60
  const display = `${minutes}:${String(seconds).padStart(2, '0')}`

  return (
    <div className={`flex flex-col items-center gap-2 py-3 ${isWarning ? 'animate-pulse' : ''}`}>
      {/* Countdown display */}
      {done ? (
        <div className="flex items-center gap-2 text-primary animate-bounce" style={{ animationDuration: '2s' }}>
          <svg width="24" height="24" viewBox="0 0 16 16" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round"><path d="M3 8l4 4 6-7"/></svg>
          <span className="text-xl font-bold">Done</span>
        </div>
      ) : (
        <span className={`text-3xl font-bold tabular-nums ${isWarning ? 'text-amber-500' : 'text-text'}`}>
          {display}
        </span>
      )}

      {/* Controls row */}
      <div className="flex gap-2 items-center">
        {!running && !done && (
          <button
            onClick={onStart}
            className="bg-primary text-white rounded-[--radius-btn] px-4 py-2 text-sm font-semibold"
          >
            Start
          </button>
        )}
        {running && (
          <button
            onClick={onPause}
            className="bg-primary text-white rounded-[--radius-btn] px-4 py-2 text-sm font-semibold"
          >
            Pause
          </button>
        )}
        {!running && secondsRemaining > 0 && secondsRemaining < (minutes * 60 + seconds) && (
          <button
            onClick={onReset}
            className="bg-primary text-white rounded-[--radius-btn] px-4 py-2 text-sm font-semibold"
          >
            Reset
          </button>
        )}

        {/* Skip */}
        {!done && (
          <>
            {showSkipConfirm ? (
              <div className="flex items-center gap-2">
                <span className="text-xs text-text/60">Skip the rest of this wait?</span>
                <button
                  onClick={() => { setShowSkipConfirm(false); onSkip() }}
                  className="text-xs text-primary font-semibold"
                >
                  Yes
                </button>
                <button
                  onClick={() => setShowSkipConfirm(false)}
                  className="text-xs text-text/60"
                >
                  No
                </button>
              </div>
            ) : (
              <button
                onClick={() => setShowSkipConfirm(true)}
                className="text-text/60 hover:text-text text-xs underline"
              >
                Skip
              </button>
            )}
          </>
        )}
      </div>
    </div>
  )
}
```
