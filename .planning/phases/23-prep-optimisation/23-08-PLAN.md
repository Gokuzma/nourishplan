  })

    expect(parseStepsSafely({ steps: [] })).toBeNull()
  })

  it('returns null for non-array input (number)', () => {

  it('useRegenerateRecipeSteps is only imported in RecipeBuilder', () => {
    // Grep all .tsx/.ts files for useRegenerateRecipeSteps imports
    const srcDir = path.resolve('src')
    const importingFiles: string[] = []

    function walk(dir: string) {
      for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
        const full = path.join(dir, entry.name)
        if (entry.isDirectory()) {
          walk(full)
        } else if (entry.name.endsWith('.ts') || entry.name.endsWith('.tsx')) {
          const content = fs.readFileSync(full, 'utf8')
          // Check for import of useRegenerateRecipeSteps (not its definition)
          if (content.includes('useRegenerateRecipeSteps') && !full.includes('useRecipeSteps.ts')) {
            importingFiles.push(full.replace(srcDir + path.sep, ''))
          }
        }
      }
    }
    walk(srcDir)

    // Should only appear in RecipeBuilder.tsx (the ingredient save handler)
    // and possibly in the hook definition file itself
    const consumers = importingFiles.filter(f => !f.includes('hooks/'))
    expect(consumers).toEqual(
      expect.arrayContaining([expect.stringContaining('RecipeBuilder')])
    )
    // No other component should import it
    const nonBuilderConsumers = consumers.filter(f => !f.includes('RecipeBuilder'))
    expect(nonBuilderConsumers).toHaveLength(0)
  })
})

import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest'
import { render, screen, fireEvent } from '@testing-library/react'

describe('NotificationPermissionBanner (V-09)', () => {
  const mockRequest = vi.fn()

  beforeEach(() => {
    vi.clearAllMocks()
    localStorage.clear()
  })

  afterEach(() => {
    localStorage.clear()
  })

  it('renders prompt when permission is default', () => {
    vi.mocked(useNotificationPermission).mockReturnValue({
      permission: 'default',
      request: mockRequest,
    })

    render(<NotificationPermissionBanner onPermissionChange={vi.fn()} />)
    expect(screen.getByText('Get notified when timers finish')).toBeTruthy()
    expect(screen.getByText('Allow notifications')).toBeTruthy()
    expect(screen.getByText('Not now')).toBeTruthy()
  })

  it('renders denied state when permission is denied', () => {
    vi.mocked(useNotificationPermission).mockReturnValue({
      permission: 'denied',
      request: mockRequest,
    })

    render(<NotificationPermissionBanner onPermissionChange={vi.fn()} />)
    expect(screen.getByText(/Notifications blocked/)).toBeTruthy()
  })

  it('does not render when permission is granted', () => {
    vi.mocked(useNotificationPermission).mockReturnValue({
      permission: 'granted',
      request: mockRequest,
    })

    const { container } = render(<NotificationPermissionBanner onPermissionChange={vi.fn()} />)
    expect(container.innerHTML).toBe('')
  })

  it('hides after Not now is clicked and sets localStorage cooldown', () => {
    vi.mocked(useNotificationPermission).mockReturnValue({
      permission: 'default',
      request: mockRequest,
    })

    render(<NotificationPermissionBanner onPermissionChange={vi.fn()} />)
    fireEvent.click(screen.getByText('Not now'))

    const stored = localStorage.getItem('cook-notification-dismissed-at')
    expect(stored).toBeTruthy()
    expect(Number(stored)).toBeGreaterThan(Date.now() - 5000)
  })

  it('does not render when cooldown is active', () => {
    localStorage.setItem('cook-notification-dismissed-at', String(Date.now()))

    vi.mocked(useNotificationPermission).mockReturnValue({
      permission: 'default',
      request: mockRequest,
    })

    const { container } = render(<NotificationPermissionBanner onPermissionChange={vi.fn()} />)
    expect(container.innerHTML).toBe('')
  })

  it('renders again after 7-day cooldown expires', () => {
    const eightDaysAgo = Date.now() - (8 * 24 * 60 * 60 * 1000)
    localStorage.setItem('cook-notification-dismissed-at', String(eightDaysAgo))

    vi.mocked(useNotificationPermission).mockReturnValue({
      permission: 'default',
      request: mockRequest,
    })

    render(<NotificationPermissionBanner onPermissionChange={vi.fn()} />)
    expect(screen.getByText('Get notified when timers finish')).toBeTruthy()
  })
})
